#### debounce防抖

```js
function debounce(fn,delay=500){
    let timer = null;
    return function(){
        if (timer){
            clearTimeout(timer);
        }
        timer = setTimeout(()=>{
            fn.apply(this,arguments);//apply的话调用函数不能为箭头函数
            //fn()也可以
            timer = null;
        },delay);
    }
}
```

#### 节流

```js
function throttle(fn, delay=100){
    let timer = null;
    return function(){
        if(timer){
            return
        }
        timer = setTimeout(()=>{
            fn.apply(this,arguments)
            timer = null
        },delay)
    }

}
```
